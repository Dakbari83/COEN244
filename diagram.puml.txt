  Package::calculateCost() returns weight * cost
  (weight and cost validated to be positive in ctor)

' TwoDayPackage derived class
class TwoDayPackage {
  - fee : double

  + TwoDayPackage()
  + TwoDayPackage(..., f: double)
  + TwoDayPackage(other: TwoDayPackage&)

  + ~TwoDayPackage() : default
  + calculateCost() : double <<override, final>>
}

note right of TwoDayPackage
  TwoDayPackage::calculateCost() = Package::calculateCost() + fee
end note

' OvernightPackage derived class
class OvernightPackage {
  - fee : double   ' additional fee per ounce

  + OvernightPackage()
  + OvernightPackage(..., f: double)
  + OvernightPackage(other: OvernightPackage&)

  + ~OvernightPackage() : default
  + calculateCost() : double <<override, final>>
}

note right of OvernightPackage
  OvernightPackage::calculateCost() = (cost + fee) * weight
  (fee is additional per-ounce charge)
end note
